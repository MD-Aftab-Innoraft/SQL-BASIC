CREATE TABLE fixtures (
match_id INT PRIMARY KEY AUTO_INCREMENT,
venue varchar(20),
date DATE,
team_1 varchar(35) references teams(team_name),
team_2 varchar(35) references teams(team_name),
toss_won_by varchar(35) references teams(team_name),
match_won_by varchar(35) references teams(team_name));
